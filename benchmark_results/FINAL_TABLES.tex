% KEY STATISTICS BOX
% ============================================================================
\begin{tcolorbox}[colback=blue!5,colframe=blue!50!black,title=\textbf{Key Findings}]
\begin{itemize}[nosep,leftmargin=*]
\item Multi-agent systems leak \textbf{2.3$\times$} more than single-agent (37\% vs 16\%)
\item Internal channels (C2, C5) have \textbf{8.3$\times$} higher leak rate than external
\item Sanitizer is \textbf{98\% effective on C1} but \textbf{0\% on C2/C5}
\item \textbf{All major frameworks} lack inter-agent privacy mechanisms
\end{itemize}
\end{tcolorbox}
